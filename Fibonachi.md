│
├── fib(5)

│   │   │   │   └── fib(0) = 0
│   │   │   = 1 + 0 = 1
│   │   │   └── fib(1) = 1
│   │   = 1 + 1 = 2
│   │   └── fib(2)
│   │       ├── fib(1) = 1
│   │       └── fib(0) = 0
│   │   = 1 + 0 = 1
│   = 2 + 1 = 3
│   └── fib(3)
│       ├── fib(2)

│       = 1 + 0 = 1
│       └── fib(1) = 1
│   = 1 + 1 = 2
= 3 + 2 = 5
└── fib(4)
    ├── fib(3)
    │   ├── fib(2)
    │   │   ├── fib(1) = 1
    │   │   └── fib(0) = 0
    │   = 1 + 0 = 1
    │   └── fib(1) = 1
    = 1 + 1 = 2
    └── fib(2)
        ├── fib(1) = 1
        └── fib(0) = 0
    = 1 + 0 = 1
= 2 + 1 = 3

fib(6) = fib(5) + fib(4) = 5 + 3 = 8
